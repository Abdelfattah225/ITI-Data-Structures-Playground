int find(V){
    if(root[V] != v){
        root[v] = find(root[v])       
    }
    return root[v]
}

int union(u,v){
    rootu = find(u);
    rootv = find(v);
    if(rootu == rootv){
        return 1;
    }
    /// merge ///
    parent[rootu] = rootv;
}
